Future<OkMessage> sendEvent(EventMessage event) async {
  try {
    sendMessage(event);
    return messages
        .where((message) => message is OkMessage)
        .cast<OkMessage>()
        .firstWhere((message) => message.eventId == event.id);
  } catch (error, stack) {
    logWarning("$url Failed to send event $event", error, stack);
    rethrow;
  }
}
